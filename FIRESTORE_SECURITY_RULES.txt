// Firestore Security Rules for Notifications
// Copy this to Firebase Console -> Firestore Database -> Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow authenticated users to read/write their own user document
    match /users/{userId} {
      allow read, write: if request.auth.uid == userId;
      allow read: if request.auth != null; // Needed for officials query
      
      // User's notifications subcollection
      match /notifications/{notification} {
        allow read: if request.auth.uid == userId;
        allow create: if request.auth != null; // Allow any authenticated user to create notifications
        allow update, delete: if request.auth.uid == userId;
      }
    }
    
    // Proposals collection - allow all authenticated users to read/write
    match /proposals/{proposal} {
      allow read, write: if request.auth != null;
    }

    // Allow reads for public data (if needed)
    match /{document=**} {
      allow read: if false;
      allow write: if false;
    }
  }
}
